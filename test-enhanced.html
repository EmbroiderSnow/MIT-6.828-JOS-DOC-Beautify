/**
 * 简单的ELF加载器函数
 */
static void
readseg(uchar* pa, uint count, uint offset)
{
    uchar* epa;
    
    epa = pa + count;
    
    // round down to sector boundary
    pa -= offset % SECTSIZE;
    
    // translate from bytes to sectors, and kernel starts at sector 1
    offset = (offset / SECTSIZE) + 1;
}

汇编代码示例

.globl _start
_start:
    # 切换到保护模式
    lgdt gdtdesc
    movl %cr0, %eax
    orl $CR0_PE_ON, %eax
    movl %eax, %cr0
    
    # 跳转到下一条指令，但在32位代码段中
    ljmp $(SEG_KCODE<<3), $protcseg

protcseg:
    # 设置保护模式数据段寄存器
    movw $(SEG_KDATA<<3), %ax
    movw %ax, %ds
    movw %ax, %es
    movw %ax, %fs
    movw %ax, %gs
    movw %ax, %ss
